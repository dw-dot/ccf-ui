export interface ColorScheme {

  /**
   * Whether the scheme is discrete or gradient
   */
  type: 'discrete' | 'gradient';

  /**
   * Name of scheme
   */
  name: string;

  /**
   * Colors used in the scheme
   */
  colors: string[];

  /**
   * Positions for mapping the data to colors
   */
  positions: number[];
}
