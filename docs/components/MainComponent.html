import { Component } from '@angular/core';

import { NavigationService } from '../../shared/services/navigation/navigation.service';
import { SearchService } from '../../shared/services/search/search.service';

/**
 * Main view containing common items to all subviews.
 */
@Component({
  selector: 'ccf-main',
  templateUrl: './main.component.html',
  styleUrls: ['./main.component.scss']
})
export class MainComponent {
  /**
   * Indicates whether the search drawer is active(open)/inactive(closed).
   */
  searchActive = true;

  /**
   * Creates an instance of main component.
   *
   * @param navigation The navigation service used to determine active routes.
   * @param search The search service used to fetch a description of the current search.
   */
  constructor(
    readonly navigation: NavigationService,
    readonly search: SearchService
  ) { }
}

<div class="ccf-main wrapper">
  <div class="container">
    <ccf-header class="header"></ccf-header>

    <div class="contents-area">
      <ccf-navigation class="navigation" [(searchActive)]="searchActive"></ccf-navigation>

      <mat-drawer-container class="drawer">
        <mat-drawer class="search" mode="side" [opened]="searchActive">
          <ccf-search></ccf-search>
        </mat-drawer>

        <mat-drawer-content class="contents">
          <div *ngIf="!(navigation.isHomeActive | async)" class="top-bar">
            <div class="search-criteria">{{ search.searchCriteria | async }}</div>
            <a rel="noreferrer noopener" href="https://sampledata.hubmapconsortium.org/home" target="_blank">
              <mat-icon class="icon font-icon download">cloud_download</mat-icon>
            </a>
          </div>
          <div class="outlet">
            <router-outlet></router-outlet>
          </div>
        </mat-drawer-content>
      </mat-drawer-container>
    </div>
  </div>
</div>
