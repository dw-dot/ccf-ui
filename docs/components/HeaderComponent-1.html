import { ChangeDetectionStrategy, Component, HostBinding } from '@angular/core';

 * Page header

  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush

  /** HTML class name */

}
