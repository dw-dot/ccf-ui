import { Component } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';

/**
 * Component for about modal, that gives brief description of the application.
 */
@Component({
  selector: 'ccf-about',
  templateUrl: './about.component.html',
  styleUrls: ['./about.component.scss']
})
export class AboutComponent {
  /**
   * Creates an instance of about component.
   * @param aboutModalService Modal service responsible for opening and closing of the modal.
   */
  constructor(private readonly aboutModal: MatDialog) { }

  /**
   * Closes About modal.
   */
  close(): void {
    this.aboutModal.closeAll();
  }
}

          The HuBMAP common coordinate (reference) framework is under active development. It aims to spatially annotate and index

          CCF Ontology documentation is available <a rel="noopener noreferrer" href="https://docs.google.com/document/d/1X21O5DgGkq9ngPOsBZa-qy1-6Y2MiohJD7Bt-JFyysY/edit" target="_blank">here</a>

        <h2 class="heading">CCF Exploration User Interface (EUI)</h2>
        <p class="description">
          The CCF EUI represents data across multiple scales, supporting navigation by multiple coordinate systems,
          including spatial, physiologic, and organ-based browsing. The first proof-of-concept user interface for the CCF
          v0.5.0 became available in June 2019 and continues to evolve as more data becomes available. CCF EUI specification is
          available <a rel="noopener noreferrer" href="https://drive.google.com/drive/folders/1K8vEcxLLOjho4zD2V4S1WFP1CX4_e30P" target="_blank">here</a>
